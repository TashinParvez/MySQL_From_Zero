    SELECT 
        'FIRST HALF of 1987 Hire Emp' AS date_RANGE,            ///-----> Puring value in the column 
        sum(salary) AS Total_Salary,
        COUNT(DISTINCT department_id) AS depts
    FROM employees
    WHERE YEAR(hire_date)= 1987 AND MONTH(hire_date) <= 6;
